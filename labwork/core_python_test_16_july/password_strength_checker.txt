import re
pwd = input("Enter password: ")
conditions = []

if len(pwd) < 8: conditions.append("Minimum 8 characters")
if not re.search(r"[A-Z]", pwd): conditions.append("At least one uppercase")
if not re.search(r"[a-z]", pwd): conditions.append("At least one lowercase")
if not re.search(r"[0-9]", pwd): conditions.append("At least one digit")
if not re.search(r"[^A-Za-z0-9]", pwd): conditions.append("At least one special character")

if conditions:
    print("Weak Password. Missing:", conditions)
else:
    print("Strong Password")
